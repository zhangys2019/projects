Themes

df <- data.frame(x = 1:20,
                 y = round(runif(20, 5, 10)),
                 z = round(runif(20, 7, 12)),
                 w = round(runif(20, 9, 14)))

plot <- df %>% 
  echart(x, width = "100%") %>% 
  earea(y, stack = "grp") %>% 
  earea(z, stack = "grp") %>% 
  earea(w, stack = "grp") %>% 
  etoolbox_magic(type = list("bar", "line", "stack", "tiled"))
