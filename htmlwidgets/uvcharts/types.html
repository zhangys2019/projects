Chart types

# create data
df <- data.frame(x = 1:10, y = runif(10, 0, 5), z = runif(10, 0, 5))

Bar

df %>%
  uv_bar(x) %>%
  uv_add(y) %>%
  uv_add(z) %>%
  uv_frame(bgcolor = "none")

Stacked bar

df %>%
  uv_stackbar(x) %>%
  uv_add(y) %>%
  uv_add(z) %>%
  uv_frame(bgcolor = "none")

Percent bar

df %>%
  uv_percentbar(x) %>%
  uv_add(y) %>%
  uv_add(z) %>%
  uv_frame(bgcolor = "none")

Stepup bar

df %>%
  uv_stepupbar(x) %>%
  uv_add(y) %>%
  uv_add(z) %>%
  uv_frame(bgcolor = "none")

Line

df %>%
  uv_line(x) %>%
  uv_add(y) %>%
  uv_add(z) %>%
  uv_graph(orientation = "Vertical") %>%
  uv_frame(bgcolor = "none")

Area

df %>%
  uv_area(x) %>%
  uv_add(y) %>%

Pie

df <- data.frame(x = c("apple", "banana", "cake", "mango", "strawberry"), y = c(1,4,2,5,3))
df %>%
  uv_pie(x) %>%
  uv_add(y) %>%
  uv_frame(bgcolor = "none")

df %>%
  uv_polar(x) %>%

  uv_frame(bgcolor = "none")

Waterfall

df %>%

  uv_frame(bgcolor = "none")

Any

Pass any charts with uv_chart specifying type (CamelCase).

df %>%
  uv_charts(x, type = "Bar") %>%
  uv_add(y) %>%
  uv_frame(bgcolor = "none")
